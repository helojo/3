    /**
     * 将文件转换成byte数组
     * @param tradeFile
     * @return
     */
    public byte[] fileToByte(String fileUrl){
        // 第1步、使用File类找到一个文件
        File f= new File("fileUrl") ;    // 声明File对象
        // 第2步、通过子类实例化父类对象
        InputStream input = new FileInputStream(f) ; 　　//实例化文件输入流
        // 第3步、进行读操作
        byte result[] = new byte[1024] ;        // 存储读取内容
        input.read(result) ;        // 读取内容 
        // 第4步、关闭输出流
        input.close() ;                        // 关闭输出流
        System.out.println("byte数组：" + result);
    　　 return result;
    }
