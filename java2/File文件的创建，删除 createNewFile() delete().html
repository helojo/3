    package seday03;

    import java.io.File;
    import java.io.IOException;

    /**
    * 使用File新建一个test1.txt文件
    * @author xingsir
    */
    public class CreateNewFileDemo {

    public static void main(String[] args) throws IOException {
    File file =new File("./test1.txt");
    if(!file.exists()) {
    file.createNewFile();
    System.out.println("已创建文件");
    }else {
    System.out.println("已存在文件");
    }
    }

    }

    package seday03;

    import java.io.File;

    /**
    * 删除一个文件
    * @author xingsir
    */
    public class DeleteFileDmeo {

    public static void main(String[] args) {
    /*
    * 删除当前目录下的test1.txt文件
    */
    File file = new File("./test1.txt");
    if(file.exists()) {
    file.delete();
    System.out.println("已删除文件");
    }else {
    System.out.println("不存在文件");
    }

    }

    }
