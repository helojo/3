    [abc]：包含abc单个字母。

    [^abc]: 不包含abc单个字母。

     

    String regex = "[abc]";

    String regex = "[^abc]";

    String regex = "[a-zA-Z]";

     

     String regex = "\\d";

    String regex = "\\D";

    String regex = "\\s";

    String regex = "\\S";

    String regex = "\\w";

    String regex = "\\W";

     

     String regex = "[abc]?";

     String regex = "[abc]*";

      String regex = "[abc]+";

     String regex = "[abc]{5}";

     String regex = "[abc]{5,}";

     String regex = "[abc]{5,9}";
